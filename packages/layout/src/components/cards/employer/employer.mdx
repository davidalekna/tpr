<Playground>
	{() => {
		const [complete, setComplete] = useState(false);
		const sleep = (ms) => new Promise((res) => setTimeout(res, ms));
		const employer = {};
		return (
			<Employer
				type="employer"
				onSave={(employer) => sleep(1000).then(() => console.log(employer))}
				onRemove={(employer) => sleep(2500).then(() => console.log(employer))}
				onCorrect={(value) => setComplete(value)}
				complete={complete}
				employer={employer}
			/>
		);
	}}
</Playground>
